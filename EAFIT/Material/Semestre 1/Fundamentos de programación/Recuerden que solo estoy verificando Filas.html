import java.util.*;
import java.math.*;

class Animal
{
String tipo;
String nombre;
String sonido;

public Animal(String miTipo, String miNombre, String miSonido)
{
tipo=miTipo;
nombre=miNombre;
sonido=miSonido;
}

public String Sonido()
{
return sonido;
}

}


class Main
{
static enum Mode
{
Columna,
Fila,
DiagonalPrimaria,
DiagonalSecundaria
}

public static void main(String[] args)
{
String sonidoPato="Quack quack";
String sonidoGato= "Miau miau";

Scanner consola = new Scanner(System.in);
System.out.print("Matriz cuadrada, ingresa la base: ");
int n = consola.nextInt();
Animal[][] matriz= new Animal[n][n];

int patos= 0;
int fila=0;
int columna=0;

//Animal Pato= new Animal("Pato","Donald",sonidoPato);
//Animal Gato= new Animal("Gato","Tom",sonidoGato);

FillMatrix(matriz,n,sonidoGato);

while(patos<n)
{
fila=(int)(Math.random()*n);
columna=(int)(Math.random()*n);
if(matriz[fila][columna].Sonido()!=sonidoPato)
{
matriz[fila][columna]=new Animal("Pato","Donald",sonidoPato);
patos++;
}
}

ImprimirMatrix(matriz,n);

System.out.println();
System.out.println(CheckingTrick(matriz, n, sonidoPato, Mode.Fila));

}

public static void FillMatrix(Animal[][] matrix, int n, String sonidoGato)
{
for(int i=0; i<n;i++)
{
for(int j=0;j<n;j++)
{
matrix[i][j]=new Animal("Gato","Tom",sonidoGato);
}
}
}

public static void ImprimirMatrix(Animal[][] matrix, int n)
{
for(int i=0; i<n;i++)
{
for(int j=0;j<n;j++)
{
System.out.print("\t" +matrix[i][j].Sonido());
}

System.out.println();
}
}

public static boolean CheckingTrick(Animal[][] matrix, int n, String sonidoPato, Mode myMode )
{
boolean result=true;
switch(myMode)
{
case Fila:
{
for(int i=0;i<n;i++)
{
result=true;
for(int j=0;j<n;j++)
{
if(matrix[i][j].Sonido()!=sonidoPato)
{
result=false;
break;
}
}
if (result)
break;
}
if(result)
{
return true;
}
else
return false;
}
default:
return false;
}
}
}
